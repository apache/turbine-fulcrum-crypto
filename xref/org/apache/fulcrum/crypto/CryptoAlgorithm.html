24  /**
25   * This interface describes the various Crypto Algorithms that are
26   * handed out by the Crypto Service.
27   *
28   * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
29   * @version $Id$
30   */
31  public interface CryptoAlgorithm
32  {
33      /**
34       * Allows the user to set a salt value whenever the
35       * algorithm is used. Setting a new salt should invalidate
36       * all internal state of this object.
37       * <p>
38       * Algorithms that do not use a salt are allowed to ignore
39       * this parameter.
40       * <p>
41       * Algorithms must be able to deal with the null value as salt.
42       * They should treat it as "use a random salt".
43       *
44       * @param salt      The salt value
45       *
46       */
47  
48      void setSeed(String salt);
49  
50      /**
51       * Performs the actual encryption.
52       *
53       * @param value       The value to be encrypted
54       *
55       * @return The encrypted value
56       *
57       * @throws Exception various errors from the underlying ciphers.
58       *                   The caller should catch them and report accordingly.
59       *
60       */
61  
62      String encrypt(String value)
63          throws Exception;
64  
65      /**
66       * Algorithms that perform multiple ciphers get told
67       * with setCipher, which cipher to use. This should be
68       * called before any other method call.
69       *
70       * If called after any call to encrypt or setSeed, the
71       * CryptoAlgorithm may choose to ignore this or to reset
72       * and use the new cipher.
73       *
74       * If any other call is used before this, the algorithm
75       * should use a default cipher and not throw an error.
76       *
77       * @param cipher    The cipher to use.
78       *
79       */
80  
81      void setCipher(String cipher);
82  
83  }
